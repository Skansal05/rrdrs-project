process_shipment() Procedure
PROCEDURE CREATE OR REPLACE PROCEDURE process_shipment (
  p_request_id NUMBER,
  p_warehouse_id NUMBER,
  p_quantity NUMBER
) IS
BEGIN
  INSERT INTO Shipments
    (shipment_id, request_id, warehouse_id,
     dispatch_date, shipped_quantity, status)
  VALUES (seq_shipment.NEXTVAL,
    p_request_id, p_warehouse_id,
    SYSDATE, p_quantity, 'In Transit');

  UPDATE Relief_Requests
  SET status = 'Fulfilled'
  WHERE request_id = p_request_id;

  COMMIT;
END;

① trg_no_negative_stock
BEFORE INSERT OR UPDATE
Validates that quantity_on_hand is never set below 0. Uses RAISE_APPLICATION_ERROR to reject the DML with ORA-20001.
TRIGGER 1 / 5 CREATE OR REPLACE TRIGGER trg_no_negative_stock
BEFORE INSERT OR UPDATE ON Camp_Inventory
FOR EACH ROW
BEGIN
  IF :NEW.quantity_on_hand < 0 THEN
    RAISE_APPLICATION_ERROR(-20001,
      'Stock cannot be negative');
  END IF;
END;

③ trg_low_stock_alert
AFTER UPDATE OF quantity_on_hand · WHEN < 10
The core automation trigger. When stock drops below 10 units, it auto-inserts a Critical priority relief request using seq_request.NEXTVAL. This fires the subsequent trg_set_priority trigger on that INSERT.
TRIGGER 3 / 5 CREATE OR REPLACE TRIGGER trg_low_stock_alert
AFTER UPDATE OF quantity_on_hand
ON Camp_Inventory
FOR EACH ROW
WHEN (NEW.quantity_on_hand < 10)
BEGIN
  INSERT INTO Relief_Requests (
    request_id, camp_id, item_id,
    quantity_requested, request_date,
    status, priority_level
  ) VALUES (
    seq_request.NEXTVAL,
    :NEW.camp_id, :NEW.item_id,
    50, SYSDATE,
    'Pending', 'Critical'
  );
END;
/
④ trg_set_priority
BEFORE INSERT ON Relief_Requests
Overrides the priority_level field on every new request based on quantity thresholds. Ensures consistent priority assignment regardless of what the user submits.
TRIGGER 4 / 5 CREATE OR REPLACE TRIGGER trg_set_priority
BEFORE INSERT ON Relief_Requests
FOR EACH ROW
BEGIN
  IF :NEW.quantity_requested > 100 THEN
    :NEW.priority_level := 'Critical';
  ELSIF :NEW.quantity_requested > 50 THEN
    :NEW.priority_level := 'High';
  ELSIF :NEW.quantity_requested > 20 THEN
    :NEW.priority_level := 'Medium';
  ELSE
    :NEW.priority_level := 'Low';
  END IF;
END;
/
⑤ trg_update_inventory_on_delivery
AFTER UPDATE OF status · WHEN = 'Delivered'
When a shipment is marked Delivered, this trigger auto-adds shipped_quantity to the camp's Camp_Inventory by joining through Relief_Requests to find the correct (camp_id, item_id) pair.
TRIGGER 5 / 5 CREATE OR REPLACE TRIGGER trg_update_inventory_on_delivery
AFTER UPDATE OF status ON Shipments
FOR EACH ROW
WHEN (NEW.status = 'Delivered')
BEGIN
  UPDATE Camp_Inventory
  SET quantity_on_hand = quantity_on_hand + :NEW.shipped_quantity
  WHERE camp_id = (
    SELECT camp_id FROM Relief_Requests
    WHERE request_id = :NEW.request_id
  )
  AND item_id = (
    SELECT item_id FROM Relief_Requests
    WHERE request_id = :NEW.request_id
  );
END;
/
process_shipment() Procedure
PROCEDURE
Encapsulates the full shipment dispatch flow: inserts into Shipments, marks the request Fulfilled, and commits — all in one atomic call.
PROCEDURE CREATE OR REPLACE PROCEDURE process_shipment (
  p_request_id NUMBER,
  p_warehouse_id NUMBER,
  p_quantity NUMBER
) IS
BEGIN
  INSERT INTO Shipments (
    shipment_id, request_id, warehouse_id,
    dispatch_date, delivery_date,
    shipped_quantity, status
  ) VALUES (
    seq_shipment.NEXTVAL,
    p_request_id, p_warehouse_id,
    SYSDATE, NULL,
    p_quantity, 'In Transit'
  );

  UPDATE Relief_Requests
  SET status = 'Fulfilled'
  WHERE request_id = p_request_id;

  COMMIT;
END;
/

-- Execute:
BEGIN
  process_shipment(1, 1, 50);
END;

get_days_left() Function
FUNCTION → NUMBER
Returns days of supply remaining for a given (camp_id, item_id) pair. Uses a 3-table join to compute quantity ÷ (population × per_person_daily_need). Called directly in SQL queries.
FUNCTION CREATE OR REPLACE FUNCTION get_days_left (
  p_camp_id NUMBER,
  p_item_id NUMBER
) RETURN NUMBER IS
  v_days NUMBER;
BEGIN
  SELECT
    ci.quantity_on_hand /
    (rc.current_population * sc.per_person_daily_need)
  INTO v_days
  FROM Camp_Inventory ci
  JOIN Relief_Camps rc ON ci.camp_id = rc.camp_id
  JOIN Supply_Catalog sc ON ci.item_id = sc.item_id
  WHERE ci.camp_id = p_camp_id
  AND ci.item_id = p_item_id;

  RETURN ROUND(v_days, 2);
END;
/

-- Usage:
SELECT get_days_left(1, 101) FROM dual;

OPTIMIZED QUERY -- Critical shortage report — 3-table join, calculated column
SELECT
  rc.camp_name,
  rc.district,
  sc.item_name,
  ci.quantity_on_hand,
  ROUND(
    ci.quantity_on_hand /
    (rc.current_population * sc.per_person_daily_need), 2
  ) AS days_left,
  rc.current_population
FROM Camp_Inventory ci
JOIN Relief_Camps rc ON ci.camp_id = rc.camp_id
JOIN Supply_Catalog sc ON ci.item_id = sc.item_id
WHERE
  (ci.quantity_on_hand /
  (rc.current_population * sc.per_person_daily_need)) < 3
ORDER BY days_left ASC;
